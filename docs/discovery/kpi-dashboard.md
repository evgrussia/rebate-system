## 7. Alert Rules (Grafana)

### Critical Alerts (Telegram + Email)

```yaml
- name: Service Health
  rules:
    - alert: ServiceDown
      expr: up{job="cryptorebate"} == 0
      for: 2m
      severity: critical

    - alert: HighErrorRate
      expr: rate(django_http_exceptions_total[5m]) / rate(django_http_requests_total[5m]) > 0.05
      for: 5m
      severity: critical

    - alert: PayoutFailureSpike
      expr: rate(payout_requests_total{status="failed"}[1h]) > 0.05
      for: 10m
      severity: critical
```

### Warning Alerts (Telegram)

```yaml
- name: Performance
  rules:
    - alert: SlowAPI
      expr: histogram_quantile(0.95, django_http_request_duration_seconds_bucket) > 2
      for: 5m
      severity: warning

    - alert: HighCPU
      expr: 100 - (avg(rate(node_cpu_seconds_total{mode="idle"}[5m])) * 100) > 85
      for: 10m
      severity: warning

    - alert: HighMemory
      expr: (1 - node_memory_MemAvailable_bytes / node_memory_MemTotal_bytes) > 0.9
      for: 5m
      severity: warning

    - alert: DiskLow
      expr: (1 - node_filesystem_avail_bytes / node_filesystem_size_bytes) > 0.85
      for: 5m
      severity: warning

    - alert: ExchangeSyncStale
      expr: time() - exchange_last_sync_timestamp > 7200
      for: 5m
      severity: warning
      labels:
        exchange: "{{ $labels.exchange }}"

    - alert: PayoutQueueBacklog
      expr: payout_queue_size{status="pending"} > 50
      for: 15m
      severity: warning
```
